#define E 0.00000001 // constant which represents the degree of accuracy of the square root approximation 

/* function prototype */

/**
 * Compute the approximate square root of a nonnegative number, N, using the Babylonian Method.
 * 
 * Assume that N is a float type value and that output is an output stream object.
 * 
 * This function returns a value whose data type is long double (which is a floating-point number).
 */
long double compute_square_root_of_nonnegative_integer(float N, std::ostream & output)
{
    int i = 0;
    long double X = 0.0, Y = 1.0;
    X = ((N < 0) || (N > MAXIMUM_N)) ? 0 : N;
    X = N;
    output << "\n\nN = " << N << ". // number to take the square root of";
    output << "\nX = " << X << ". // approximate square root of N";
    output << "\nY = " << Y << ". // number to add to X to divide by 2 per square root approximation";
    while (X - Y > E) 
    {
        X = (X + Y) / 2;
        Y = N / X;
        output << "\n\nwhile loop iteration # " << i << ":";
        output << "\nX = ((X + Y) / 2) = " << X << ".";
        output << "\nY = (N / X) = " << Y << ".";
        i += 1;
    }
    return X;
}
